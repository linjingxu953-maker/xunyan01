## 使用示例

```csharp
// 顺序执行
var chain = new ToolChainBuilder("分析任务")
    .AddStep("get_current_time")
    .AddStep("calculator", """{"expression": "2 + 2"}""")
    .AddStep("get_random_quote")
    .Build();

// 并行执行
var parallelChain = new ToolChainBuilder("并行查询")
    .AddParallelSteps("get_weather", "get_current_time", "get_random_quote")
    .Build();

// 条件执行
var conditionalChain = new ToolChainBuilder("条件链")
    .WithVariable("need_calc", true)
    .AddStep("get_current_time")
    .AddStepWithCondition("calculator", "need_calc == true", """{"expression": "1+1"}""")
    .Build();

// 执行
var executor = new ToolChainExecutor(toolRegistry);
var result = await executor.ExecuteAsync(chain);
```
